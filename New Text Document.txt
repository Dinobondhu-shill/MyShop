

create table users(
    id uuid primary key uuid_generate_v4(),
    email varchar(200) unique,
   password_hash TEXT not null,
    role not null check (role in ('customer', 'vendor', 'admin')),
    profile_name varchar(200), 
    profile_image text,
    date_of_birth date,
    gender check (gender in ('male', 'female', 'others'))
    marketing_preference check(marketing_preference in ('email', 'sms'))
    is_verified
    verified_at
    is_active
    created_at
    updated_at
)
